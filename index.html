        // Обработка сообщений от окна авторизации
        window.addEventListener('message', (event) => {
            if (event.data.type === 'authSuccess') {
                const user = event.data.user;
                updateUI(user);
            } else if (event.data.type === 'authError') {
                console.error("Error during Google Sign-In:", event.data.error);
                alert(currentLanguage === 'ru' ? 'Ошибка при входе через Google: ' + event.data.error : 'Error during Google Sign-In: ' + event.data.error);
            }
        });

        // Функция выхода
        function signOut() {
            auth.signOut().then(() => {
                username = null;
                userId = null;
                localStorage.removeItem('username');
                localStorage.removeItem('userId');
                localStorage.removeItem('userPicture');
                document.getElementById('signInButton').style.display = 'block';
                document.getElementById('signOutButton').style.display = 'none';
                document.getElementById('submitButton').style.display = 'block';
                document.getElementById('userAvatar').src = 'https://placehold.co/80x80';
                document.getElementById('userName').textContent = '';
                createRipple(event);
                loadScripts();
            }).catch(error => {
                console.error("Error during sign out:", error);
            });
        }

        // Проверка состояния авторизации
        auth.onAuthStateChanged(user => {
            if (user) {
                updateUI(user);
            } else {
                username = null;
                userId = null;
                localStorage.removeItem('username');
                localStorage.removeItem('userId');
                localStorage.removeItem('userPicture');
                document.getElementById('signInButton').style.display = 'block';
                document.getElementById('signOutButton').style.display = 'none';
                document.getElementById('submitButton').style.display = 'block';
                document.getElementById('userAvatar').src = 'https://placehold.co/80x80';
                document.getElementById('userName').textContent = '';
                loadScripts();
            }
            isAuthChecked = true;
        });

        // Обновление интерфейса
        function updateUI(user) {
            username = user.displayName;
            userId = user.uid;
            localStorage.setItem('username', username);
            localStorage.setItem('userId', userId);
            localStorage.setItem('userPicture', user.photoURL || 'https://placehold.co/80x80');
            database.ref('users/' + userId).set({
                username: username,
                picture: user.photoURL || 'https://placehold.co/80x80'
            });
            document.getElementById('signInButton').style.display = 'none';
            document.getElementById('signOutButton').style.display = 'block';
            document.getElementById('submitButton').style.display = 'block';
            document.getElementById('userAvatar').src = user.photoURL || 'https://placehold.co/80x80';
            document.getElementById('userName').textContent = username;
            loadUserScripts();
            loadScripts();
        }

        function changeLanguage() {
            currentLanguage = document.getElementById('languageSelect').value;
            localStorage.setItem('language', currentLanguage);
            updateLanguage();
        }

        function updateLanguage() {
            document.querySelectorAll('[data-ru]').forEach(element => {
                const text = element.getAttribute(`data-${currentLanguage}`);
                if (element.tagName === 'INPUT') {
                    element.placeholder = element.getAttribute(`data-${currentLanguage}-placeholder`);
                } else if (element.tagName === 'LABEL') {
                    element.textContent = text + ':';
                } else {
                    element.textContent = text;
                }
            });
            const themeSelect = document.getElementById('themeSelect');
            const languageSelect = document.getElementById('languageSelect');
            const filterSelect = document.getElementById('filterSelect');
            Array.from(themeSelect.options).forEach(option => {
                option.text = option.getAttribute(`data-${currentLanguage}`);
            });
            Array.from(languageSelect.options).forEach(option => {
                option.text = option.getAttribute(`data-${currentLanguage}`);
            });
            Array.from(filterSelect.options).forEach(option => {
                option.text = option.getAttribute(`data-${currentLanguage}`);
            });
        }

        function changeTheme() {
            currentTheme = document.getElementById('themeSelect').value;
            localStorage.setItem('theme', currentTheme);
            document.body.classList.remove('light-theme', 'space-theme');
            if (currentTheme === 'light') {
                document.body.classList.add('light-theme');
            } else if (currentTheme === 'space') {
                document.body.classList.add('space-theme');
            }
            updateThemeClasses();
        }

        function updateThemeClasses() {
            const isLight = document.body.classList.contains('light-theme');
            const isSpace = document.body.classList.contains('space-theme');
            document.querySelectorAll('.script-card, .script-info, .sidebar, .search-container, .search, .filter-select, .modal-content, .fullscreen-content, .settings-content, .history-content, .username, .modal-content input, .modal-content textarea, .settings-content select, .fullscreen-content pre, .modal-content small, .settings-content small, .history-content small').forEach(el => {
                el.classList.remove('light-theme', 'space-theme');
                if (isLight) el.classList.add('light-theme');
                if (isSpace) el.classList.add('space-theme');
            });
        }

        function openModal() {
            if (!isAuthChecked) return;
            if (!userId) {
                showNotification();
                flashSignInButton();
                return;
            }
            document.getElementById('submitModal').style.display = 'flex';
            createRipple(event);
        }

        function closeModal() {
            document.getElementById('submitModal').style.display = 'none';
        }

        function openFullscreen(index) {
            const script = scripts[index];
            document.getElementById('fullAvatar').src = script.avatar || 'https://placehold.co/200x150';
            document.getElementById('fullName').textContent = script.name || 'Без названия';
            document.getElementById('fullUser').textContent = script.user || 'Аноним';
            document.getElementById('fullDesc').textContent = script.desc || 'Нет описания';
            document.getElementById('fullTarget').textContent = `Target: ${script.target || 'Не указано'}`;
            document.getElementById('fullCode').textContent = script.code || 'Код отсутствует';
            const likeBtn = document.getElementById('fullLikeBtn');
            const dislikeBtn = document.getElementById('fullDislikeBtn');
            likeBtn.querySelector('.like-count').textContent = script.likes || 0;
            dislikeBtn.querySelector('.dislike-count').textContent = script.dislikes || 0;
            likeBtn.classList.toggle('liked', script.userRatings && script.userRatings[userId] === 'like');
            dislikeBtn.classList.toggle('disliked', script.userRatings && script.userRatings[userId] === 'dislike');
            likeBtn.onclick = () => rateScript(index, 'like', null);
            dislikeBtn.onclick = () => rateScript(index, 'dislike', null);
            document.getElementById('fullscreenWindow').style.display = 'flex';
            updateThemeClasses();
            createRipple(event);
        }

        function closeFullscreen() {
            document.getElementById('fullscreenWindow').style.display = 'none';
        }

            document.getElementById('themeSelect').value = currentTheme;
            document.getElementById('languageSelect').value = currentLanguage;

        function checkCodeLength() {
            const code = document.getElementById('scriptCode').value;
            const lines = code.split('\n').length;
            document.getElementById('lineCount').textContent = `Строк: ${lines}`;
            if (lines > 8000) {
                document.getElementById('scriptCode').value = code.split('\n').slice(0, 8000).join('\n');
                alert(currentLanguage === 'ru' ? 'Превышен лимит в 8000 строк!' : 'Exceeded 8000 line limit!');
            }
        }

        function submitScript() {
            const avatarInput = document.getElementById('avatarInput').files[0];
            const name = document.getElementById('scriptName').value;
            const desc = document.getElementById('scriptDesc').value;
            const code = document.getElementById('scriptCode').value;
            const target = document.getElementById('scriptTarget').value;

            if (!avatarInput || !name || !desc || !code || !target) {
                alert(currentLanguage === 'ru' ? 'Заполните все поля!' : 'Fill in all fields!');
                return;
            }

            const reader = new FileReader();
            reader.onload = function(e) {
                const script = { name, user: username, desc, target, code, avatar: e.target.result, likes: 0, dislikes: 0, userRatings: {} };
                const scriptId = database.ref('scripts').push().key;
                database.ref('scripts/' + scriptId).set(script);
                database.ref('userScripts/' + userId + '/' + scriptId).set(script);
                sendToTelegram(script);
                closeModal();
                openHistory();
            };
            reader.readAsDataURL(avatarInput);
            createRipple(event);
        }

        function sendToTelegram(script) {
            const message = `Новый скрипт:\nНазвание: ${script.name}\nАвтор: ${script.user}\nОписание: ${script.desc}\nЦель: ${script.target}\nКод:\n${script.code}`;
            fetch(`https://api.telegram.org/bot${TELEGRAM_BOT_TOKEN}/sendPhoto`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    chat_id: TELEGRAM_CHAT_ID,
                    photo: script.avatar,
                    caption: message
                })
            }).then(response => {
                console.log("Telegram message sent successfully:", response);
            }).catch(error => {
                console.error('Ошибка отправки в Telegram:', error);
            });
        }

        function rateScript(index, type, event) {
            if (!isAuthChecked) return;
            if (!userId) {
                showNotification();
                flashSignInButton();
                return;
            }

            const script = scripts[index];
            const scriptId = script.id;
            const previousRating = script.userRatings[userId];

            if (previousRating === type) {
                if (type === 'like') script.likes--;
                else script.dislikes--;
                delete script.userRatings[userId];
            } else {
                if (previousRating === 'like') script.likes--;
                else if (previousRating === 'dislike') script.dislikes--;
                if (type === 'like') script.likes++;
                else script.dislikes++;
                script.userRatings[userId] = type;
            }

            database.ref('scripts/' + scriptId).update({
                likes: script.likes,
                dislikes: script.dislikes,
                userRatings: script.userRatings
            });

            renderScripts();
            if (document.getElementById('fullscreenWindow').style.display === 'flex') {
                openFullscreen(index);
            }
            if (event) createRipple(event);
        }

        function loadScripts() {
            database.ref('scripts').on('value', (snapshot) => {
                scripts = [];
                const data = snapshot.val();
                if (data) {
                    Object.keys(data).forEach(key => {
                        scripts.push({ id: key, ...data[key] });
                    });
                }
                renderScripts();
            }, (error) => {
                console.error("Error loading scripts from Firebase:", error);
            });
        }

        function loadUserScripts() {
            if (!userId) return;
            database.ref('userScripts/' + userId).on('value', (snapshot) => {
                userScripts = [];
                const data = snapshot.val();
                if (data) {
                    Object.keys(data).forEach(key => {
                        userScripts.push({ id: key, ...data[key] });
                    });
                }
            }, (error) => {
                console.error("Error loading user scripts from Firebase:", error);
            });
        }

        function searchScripts() {
            const searchValue = document.getElementById('searchField').value.toLowerCase();
            const filterType = document.getElementById('filterSelect').value;
            const filteredScripts = scripts.filter(script => {
                if (filterType === 'script') {
                    return (script.name || '').toLowerCase().includes(searchValue);
                } else {
                    return (script.user || '').toLowerCase().includes(searchValue);
                }
            });
            renderScripts(filteredScripts);
        }

        function createStars() {
            const starsContainer = document.getElementById('stars');
            for (let i = 0; i < 100; i++) {
                const star = document.createElement('div');
                star.className = 'star';
                star.style.width = `${Math.random() * 3 + 1}px`;
                star.style.height = star.style.width;
                star.style.left = `${Math.random() * 100}%`;
                star.style.top = `${Math.random() * 100}%`;
                star.style.animationDuration = `${Math.random() * 5 + 5}s`;
                star.style.animationDelay = `${Math.random() * 5}s`;
                starsContainer.appendChild(star);
            }
        }

        function createRipple(event) {
            const button = event.target.closest('button');
            if (!button) return;
            const rect = button.getBoundingClientRect();
            const ripple = document.createElement('span');
            ripple.classList.add('ripple');
            const diameter = Math.max(button.clientWidth, button.clientHeight);
            const radius = diameter / 2;
            ripple.style.width = ripple.style.height = `${diameter}px`;
            ripple.style.left = `${event.clientX - rect.left - radius}px`;
            ripple.style.top = `${event.clientY - rect.top - radius}px`;
            button.appendChild(ripple);
            setTimeout(() => ripple.remove(), 600);
        }

        function showNotification() {
            const notification = document.getElementById('notification');
            notification.textContent = notification.getAttribute(`data-${currentLanguage}`);
            notification.classList.add('show');
            setTimeout(() => {
                notification.classList.remove('show');
            }, 3000);
        }

        function flashSignInButton() {
            const signInButton = document.getElementById('signInButton');
            signInButton.classList.add('flash');
            setTimeout(() => {
                signInButton.classList.remove('flash');
            }, 1000);
        }

        // Добавление обработчиков событий для кнопок
        document.querySelectorAll('.sidebar button').forEach(button => {
            button.addEventListener('click', createRipple);
        });
